CRU_convert_to_crusader_culture = { #It's here in case I want to add anatolian-themed cultures.
	if = {
		limit = {
			OR = {
				area = kerman_area
				area = mogostan_area
                area = makran_area
                area = persian_gulf_coast
			}
		}
		change_culture = crusader_baluchi
	}
	else_if = {
		limit = {
			area = kabulistan_area
            area = ghor_area
            area = balkh_area
            area = sistan_area
            area = herat_area
            area = merv_area
		}
		change_culture = crusader_afghan
	}
    else_if = {
		limit = {
		    area = iraq_arabi_area
            area = basra_area
            area = al_jazira_area
            area = shahrizor_area
		}
		change_culture = crusader_mesopotamian
	}
	else = {
		change_culture = crusader_persian
	}
}
