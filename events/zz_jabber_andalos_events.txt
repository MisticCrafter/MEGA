	desc = flavor_andalos.7.desc

	fire_only_once = yes
	
	trigger = {
		tag = ARC

		OR = {
			NOT = {
				north_america = {
					has_discovered = ROOT
				}
			}
			NOT = {
				south_america = {
					has_discovered = ROOT
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}
	
	option = {
        # Reject his proposal
		name = flavor_andalos.7.a
		ai_chance = { factor = 75 }

			limit = {
				any_known_country = {
					has_idea = quest_for_the_new_world
					religion = catholic
				}
			}
			random_known_country = {
				limit = {
					NOT = { has_country_flag = turned_down_columbus }
					religion = catholic
					has_idea = quest_for_the_new_world
