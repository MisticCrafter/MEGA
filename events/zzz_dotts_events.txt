country_event = {
	id = dotts_events.1
	title = dotts_events.1.t
	desc = dotts_events.1.d
	picture = DIPLOMACY_eventPicture
	hidden = yes
	trigger = {
		OR = {
			AND = {
				has_country_modifier = out_kni_ally

					KNI = {

				}
			}
			AND = {
				has_country_modifier = out_lae_ally
				NOT = {
					LAE = {
						alliance_with = ROOT
					}
				}
			}
			AND = {
				has_country_modifier = out_nax_ally
				NOT = {
					NAX = {
						alliance_with = ROOT
					}
				}
			}
			AND = {
				has_country_modifier = out_mor_ally
				NOT = {
					MOR = {
						alliance_with = ROOT
					}
				}
			}
			AND = {
				has_country_modifier = out_tem_ally
				NOT = {
					TEM = {
						alliance_with = ROOT
					}
				}
			}
		}
		
	}
	option = {
		if = {
			limit = {
				has_country_modifier = out_kni_ally
				NOT = {
					KNI = {
						alliance_with = ROOT
					}
				}
			}
			remove_country_modifier = out_kni_ally
		}
		if = {
			limit = {
				has_country_modifier = out_lae_ally
				NOT = {
					LAE = {
						alliance_with = ROOT
					}
				}
			}
			remove_country_modifier = out_lae_ally
		}
		if = {
			limit = {
				has_country_modifier = out_nax_ally
				NOT = {
					NAX = {
						alliance_with = ROOT
					}
				}
			}
			remove_country_modifier = out_nax_ally
		}
		if = {
			limit = {
				has_country_modifier = out_mor_ally
				NOT = {
					MOR = {
						alliance_with = ROOT
					}
				}
			}
			remove_country_modifier = out_mor_ally
		}
		if = {
			limit = {

					TEM = {
						alliance_with = ROOT
					}

		}
